usage: preprocessing.jl [-d DATE] [-f FILE]
                        [--source_data_path SOURCE_DATA_PATH]
                        [--exo_delays EXO_DELAYS] [--use_real_time]
                        [--split_transits] [--rotations] [-h]

optional arguments:
  -d, --date DATE       Date from which we want to extract the
                        timetable. (default: "25.03.19")
  -f, --file FILE       File with the timetable to process. (default:
                        "")
  --source_data_path SOURCE_DATA_PATH
                        Source of the pad zuglauf... (default:
                        "../data/hidden_data/")
  --exo_delays EXO_DELAYS
                        Number of files with exo delays to be created.
                        (type: Int64, default: 0)
  --use_real_time       Use the real time column of the timetable
                        instead of the scheduled time.
  --split_transits      Splits a Durchfahrt into two events by adding
                        a durchfahrt_out few seconds later.
  --rotations           Create a file with train dependencies. One
                        train cannot start if the other has not
                        arrived yet.
  -h, --help            show this help message and exit
